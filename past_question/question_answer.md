$$
W_q = \begin{bmatrix}
1 \\
1 \\
1
\end{bmatrix}
\in \mathbb{R}^{3 \times 1}, \quad
x_l = \begin{bmatrix} -1 \end{bmatrix}
\in \mathbb{R}^{1}
$$

$$
W_q x_l = \begin{bmatrix}
1 \cdot (-1) \\
1 \cdot (-1) \\
1 \cdot (-1)
\end{bmatrix}
= \begin{bmatrix} -1 \\ -1 \\ -1 \end{bmatrix}
\in \mathbb{R}^{3}
$$

$$
b_q = \begin{bmatrix} 0 \\ 0 \\ 0 \end{bmatrix}
\quad \Rightarrow \quad
q_l = W_q x_l + b_q
= \begin{bmatrix} -1 \\ -1 \\ -1 \end{bmatrix}
	+	\begin{bmatrix} 0 \\ 0 \\ 0 \end{bmatrix}
= \begin{bmatrix} -1 \\ -1 \\ -1 \end{bmatrix}
$$
